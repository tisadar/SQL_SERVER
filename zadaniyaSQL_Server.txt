CREATE TABLE Student (
	id INT NOT NULL AUTO_INCREMENT,
	fullname VARCHAR(200) NOT NULL,
        birthday DATE NOT NULL,
	address varchar(200) DEFAULT NULL,
	email varchar(100) DEFAULT NULL,
	group_id INT NOT NULL,
	osnova INT NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE Phone_phone_numbers (
    	id INT NOT NULL AUTO_INCREMENT,
    	student_id INT NOT NULL,
    	phone_number VARCHAR(25) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Groups (
    	id INT NOT NULL AUTO_INCREMENT,
    	group_name VARCHAR(10) NOT NULL,
    	direction_id INT NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Directions_of_study (
    	id INT NOT NULL AUTO_INCREMENT,
    	direction_name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Disciplines (
    	id INT NOT NULL IDENTITY(1,1),
    	name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Teachers (
    	id INT NOT NULL IDENTITY(1,1),
    	name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE DirectionDisciplineTeacher (
    	id INT NOT NULL IDENTITY(1,1),
    	direction_id INT NOT NULL,
    	Disciplines_id INT NOT NULL,
    	teacher_id INT NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Scores (
    	id INT NOT NULL IDENTITY(1,1),
    	student_id INT NOT NULL,
    	sub_cour_teach_id INT NOT NULL,
    	mark INT,
    	PRIMARY KEY (id)
);

CREATE TABLE Study_time (
    	id INT NOT NULL IDENTITY(1,1),
    	time_start TIME NOT NULL,
    	time_end TIME NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Lessons_shedule (
    	id INT NOT NULL IDENTITY(1,1),
    	sub_cour_teach_id INT NOT NULL,
    	time_id INT NOT NULL,
    	date DATE NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Attendance (
    	id INT NOT NULL IDENTITY(1,1),
    	tiiming_id INT NOT NULL,
    	student_id INT NOT NULL,
    	presense BIT NOT NULL,
    	PRIMARY KEY (id)
);

ALTER TABLE Student ADD CONSTRAINT Student_fk0 FOREIGN KEY (group_id) REFERENCES Groups(id);

ALTER TABLE Phone_phone_numbers ADD CONSTRAINT Phone_phone_numbers_fk0 FOREIGN KEY (student_id) REFERENCES Student(id);

ALTER TABLE Groups ADD CONSTRAINT Groups_fk0 FOREIGN KEY (direction_id) REFERENCES Directions_of_study(id);

ALTER TABLE DirectionDisciplineTeacher ADD CONSTRAINT DirectionDisciplineTeacher_fk0 FOREIGN KEY (direction_id) REFERENCES Directions_of_study(id);

ALTER TABLE DirectionDisciplineTeacher ADD CONSTRAINT DirectionDisciplineTeacher_fk1 FOREIGN KEY (Disciplines_id) REFERENCES Disciplines(id);

ALTER TABLE DirectionDisciplineTeacher ADD CONSTRAINT DirectionDisciplineTeacher_fk2 FOREIGN KEY (teacher_id) REFERENCES Teachers(id);

ALTER TABLE Scores ADD CONSTRAINT Scores_fk0 FOREIGN KEY (student_id) REFERENCES Student(id);

ALTER TABLE Scores ADD CONSTRAINT Scores_fk1 FOREIGN KEY (sub_cour_teach_id) REFERENCES DirectionDisciplineTeacher(id);

ALTER TABLE Lessons_shedule ADD CONSTRAINT Lessons_shedule_fk0 FOREIGN KEY (sub_cour_teach_id) REFERENCES DirectionDisciplineTeacher(id);

ALTER TABLE Lessons_shedule ADD CONSTRAINT Lessons_shedule_fk1 FOREIGN KEY (time_id) REFERENCES Study_time(id);

ALTER TABLE Attendance ADD CONSTRAINT Attendance_fk0 FOREIGN KEY (tiiming_id) REFERENCES Lessons_shedule(id);
